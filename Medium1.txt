//Medium2

//Single number
//Given an integer array nums where every element appears three times except for one, which appears exactly once. Find the single element and return it.


class Solution {
    public int singleNumber(int[] nums) {
        
        int one = 0;
        int two = 0;
        
        for (int i : nums)
        {
            one = (one ^ i) & (~two);
            two = (two ^ i) & (~one);
        }
        return one;
    }
}
